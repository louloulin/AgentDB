10// 智能体状态数据库 - 简化但功能完整的实现
11pub struct AgentStateDB {
12    db_path: String,
13    states: Arc<RwLock<HashMap<u64, AgentState>>>,
14    vector_states: Arc<RwLock<HashMap<u64, (AgentState, Vec<f32>)>>>,
15}

17impl AgentStateDB {
18    pub async fn new(db_path: &str) -> Result<Self, AgentDbError> {
19        // 确保数据库目录存在
20        if let Some(parent) = Path::new(db_path).parent() {
21            fs::create_dir_all(parent).map_err(|e| AgentDbError::Io(e))?;
22        }
23        
24        let db = Self {
25            db_path: db_path.to_string(),
26            states: Arc::new(RwLock::new(HashMap::new())),
27            vector_states: Arc::new(RwLock::new(HashMap::new())),
28        };
29        
30        // 尝试从磁盘加载现有数据
31        db.load_from_disk().await?;
32        
33        Ok(db)
34    }

36    async fn load_from_disk(&self) -> Result<(), AgentDbError> {
37        let states_file = format!("{}/states.json", self.db_path);
38        let vector_states_file = format!("{}/vector_states.json", self.db_path);
39        
40        // 加载普通状态
41        if Path::new(&states_file).exists() {
42            let content = fs::read_to_string(&states_file).map_err(|e| AgentDbError::Io(e))?;
43            if !content.trim().is_empty() {
44                let loaded_states: HashMap<u64, AgentState> = serde_json::from_str(&content)
45                    .map_err(|e| AgentDbError::Serialization(e.to_string()))?;
46                
47                let mut states = self.states.write().unwrap();
48                *states = loaded_states;
49            }
50        }
51        
52        // 加载向量状态
53        if Path::new(&vector_states_file).exists() {
54            let content = fs::read_to_string(&vector_states_file).map_err(|e| AgentDbError::Io(e))?;
55            if !content.trim().is_empty() {
56                let loaded_vector_states: HashMap<u64, (AgentState, Vec<f32>)> = serde_json::from_str(&content)
57                    .map_err(|e| AgentDbError::Serialization(e.to_string()))?;
58                
59                let mut vector_states = self.vector_states.write().unwrap();
60                *vector_states = loaded_vector_states;
61            }
62        }
63        
64        Ok(())
65    }

67    async fn save_to_disk(&self) -> Result<(), AgentDbError> {
68        // 确保目录存在
69        fs::create_dir_all(&self.db_path).map_err(|e| AgentDbError::Io(e))?;
70        
71        let states_file = format!("{}/states.json", self.db_path);
72        let vector_states_file = format!("{}/vector_states.json", self.db_path);
73        
74        // 使用异步写入提高性能
75        let states_json = {
76            let states = self.states.read().unwrap();
77            serde_json::to_string(&*states)
78                .map_err(|e| AgentDbError::Serialization(e.to_string()))?
79        };
80
81        let vector_states_json = {
82            let vector_states = self.vector_states.read().unwrap();
83            serde_json::to_string(&*vector_states)
84                .map_err(|e| AgentDbError::Serialization(e.to_string()))?
85        };
86
87        // 并行写入文件
88        let states_write = tokio::fs::write(&states_file, states_json);
89        let vector_states_write = tokio::fs::write(&vector_states_file, vector_states_json);
90
91        let (_, _): ((), ()) = tokio::try_join!(states_write, vector_states_write)
92            .map_err(|e| AgentDbError::Io(e))?;
93        
94        Ok(())
95    }

126    pub async fn vector_search(&self, query_embedding: Vec<f32>, limit: usize) -> Result<Vec<AgentState>, AgentDbError> {
127        let vector_states = self.vector_states.read().unwrap();
128        
129        let mut results: Vec<(f32, AgentState)> = vector_states
130            .values()
131            .map(|(state, embedding)| {
132                let similarity = self.calculate_cosine_similarity(&query_embedding, embedding);
133                (similarity, state.clone())
134            })
135            .collect();
136        
137        // 按相似度排序
138        results.sort_by(|a, b| b.0.partial_cmp(&a.0).unwrap_or(std::cmp::Ordering::Equal));
139        
140        // 返回前 limit 个结果
141        Ok(results.into_iter().take(limit).map(|(_, state)| state).collect())
142    }

144    fn calculate_cosine_similarity(&self, a: &[f32], b: &[f32]) -> f32 {
145        if a.len() != b.len() {
146            return 0.0;
147        }
148        
149        let dot_product: f32 = a.iter().zip(b.iter()).map(|(x, y)| x * y).sum();
150        let norm_a: f32 = a.iter().map(|x| x * x).sum::<f32>().sqrt();
151        let norm_b: f32 = b.iter().map(|x| x * x).sum::<f32>().sqrt();
152        
153        if norm_a == 0.0 || norm_b == 0.0 {
154            return 0.0;
155        }
156        
157        dot_product / (norm_a * norm_b)
158    }

160    pub async fn search_by_agent_and_similarity(&self, agent_id: u64, query_embedding: Vec<f32>, limit: usize) -> Result<Vec<AgentState>, AgentDbError> {
161        let vector_states = self.vector_states.read().unwrap();
162        
163        let mut results: Vec<(f32, AgentState)> = vector_states
164            .values()
165            .filter(|(state, _)| state.agent_id == agent_id)
166            .map(|(state, embedding)| {
167                let similarity = self.calculate_cosine_similarity(&query_embedding, embedding);
168                (similarity, state.clone())
169            })
170            .collect();
171        
172        // 按相似度排序
173        results.sort_by(|a, b| b.0.partial_cmp(&a.0).unwrap_or(std::cmp::Ordering::Equal));
174        
175        // 返回前 limit 个结果
176        Ok(results.into_iter().take(limit).map(|(_, state)| state).collect())
177    }

184    pub async fn delete_state(&self, agent_id: u64) -> Result<bool, AgentDbError> {
185        let removed = {
186            let mut states = self.states.write().unwrap();
187            states.remove(&agent_id).is_some()
188        };
189        
190        if removed {
191            self.save_to_disk().await?;
192        }
193        
194        Ok(removed)
195    }
